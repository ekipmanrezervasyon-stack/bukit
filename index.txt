    function apiRequest(path, opts) {
      if (!API_BASE) return Promise.reject(new Error('API base missing. Add ?api_base=https://... to URL.'));
      var token = getAuthTokenSafe_();
      if (!token) return Promise.reject(new Error('Session missing. Please sign in again.'));
      var o = opts || {};
      var method = o.method || 'GET';
      var headers = { 'Authorization': 'Bearer ' + token };
      if (method !== 'GET') headers['Content-Type'] = 'application/json';
      return fetch(API_BASE + path, {
        method: method,
        headers: headers,
        body: (method === 'GET' || o.body == null) ? undefined : JSON.stringify(o.body)
      }).then(function(res) {
        return res.text().then(function(txt) {
          var data = {};
          try { data = txt ? JSON.parse(txt) : {}; } catch (e) { data = { error: txt || 'JSON parse error' }; }
          var msg = (data && data.error) || ('HTTP ' + res.status);
          if (res.status === 401 || /unauthorized|invalid token|jwt/i.test(String(msg || ''))) {
            throw new Error('Session expired. Please sign in again.');
          }
          if (!res.ok || (data && data.ok === false)) {
            throw new Error(msg);
          }
          return data;
        });
      });
    }

    function installApiScriptRunShim_() {
      function chainRunner(methodMap) {
        var successHandler = null;
        var failureHandler = null;
        var base = {
          withSuccessHandler: function(fn) { successHandler = fn; return proxy; },
          withFailureHandler: function(fn) { failureHandler = fn; return proxy; }
        };
        function finalizeSuccess(value) {
          var cb = successHandler;
          successHandler = null;
          failureHandler = null;
          if (typeof cb === 'function') cb(value);
        }
        function finalizeFailure(err) {
          var cb = failureHandler;
          successHandler = null;
          failureHandler = null;
          if (typeof cb === 'function') cb(err);
        }
        var proxy = new Proxy(base, {
          get: function(target, prop) {
            if (Object.prototype.hasOwnProperty.call(target, prop)) return target[prop];
            return function() {
              var args = Array.prototype.slice.call(arguments);
              var fn = methodMap[String(prop)];
              Promise.resolve()
                .then(function() {
                  if (typeof fn !== 'function') throw new Error('No API mapping for ' + String(prop));
                  return fn.apply(null, args);
                })
                .then(finalizeSuccess)
                .catch(finalizeFailure);
              return undefined;
            };
          }
        });
        return proxy;
      }

    function resolveStudioCanonicalKey_(row){
      var r = row || {};
      var candidates = [r.studio_key, r.code, r.studio_code, r.name, r.id, r.studio_id];
      var map = [
        { key: 'GREEN', tokens: ['GREEN', 'YESIL', 'YEŞIL', 'YEŞİL'] },
        { key: 'RED', tokens: ['RED', 'KIRMIZI'] },
        { key: 'BLUE', tokens: ['BLUE', 'MAVI', 'MAVİ'] },
        { key: 'PODCAST', tokens: ['PODCAST'] },
        { key: 'DUBBING', tokens: ['DUBBING', 'DUBLAJ'] }
      ];
      for (var i = 0; i < candidates.length; i++) {
        var raw = String(candidates[i] || '').trim().toUpperCase();
        if (!raw) continue;
        for (var j = 0; j < map.length; j++) {
          for (var k = 0; k < map[j].tokens.length; k++) {
            if (raw === map[j].tokens[k] || raw.indexOf(map[j].tokens[k]) >= 0) return map[j].key;
          }
        }
      }
      return '';
    }

    function applyFrameGuard_() {
      try {
        if (window.top === window.self) return;
        var allowedHosts = ['script.google.com', 'script.googleusercontent.com', 'docs.google.com'];
        var refHost = '';
        try { refHost = document.referrer ? new URL(document.referrer).hostname.toLowerCase() : ''; } catch (eRef) { refHost = ''; }
        if (!refHost) return; // Referrer yoksa sessizce devam.
        var trusted = allowedHosts.some(function(h) { return refHost === h || refHost.endsWith('.' + h); });
        // Kullanici sayfasinda popup gostermeyelim; sadece sessiz bayrak koyalim.
        if (trusted) return;
        window.__bookdeckUntrustedEmbed = true;
      } catch (e) {}
    }

    function normalizePersonNameTitleCase_(raw) {
      var s = String(raw || '').trim().replace(/\s+/g, ' ');
      if (!s) return '';
      return s.split(' ').map(function(part) {
        var low = part.toLocaleLowerCase('tr-TR');
        return low.charAt(0).toLocaleUpperCase('tr-TR') + low.slice(1);
      }).join(' ');
    }

    }

        if (em && oe) oe.value = em;
        backToOtpEmailStep();

      }
      showLoginSessionRestore();
      authSessionApi_(t).then(function(res){
        var profile = res && (res.profile || res.user);
        if(!profile){
          localStorage.removeItem(AUTH_TOKEN_KEY); sessionStorage.removeItem(AUTH_TOKEN_KEY);
          showLoginOtpPanel();

          var oe = document.getElementById('otp-email');

          backToOtpEmailStep();

        currentTestToken = t;

        setTimeout(function() {
          // Warm top categories in background so first click is instant.
          prefetchEquipmentCategories_([DEFAULT_EQUIPMENT_CATEGORY, 'Sound', 'Cam', 'Ops', 'Light', 'Grip']);
        }, 250);

    function executeRecaptcha(action, cb){
      var keyMeta=document.querySelector('meta[name="recaptcha-site-key"]');
      var siteKey=keyMeta?String(keyMeta.getAttribute('content')||'').trim():'';
      if(!siteKey){ cb(''); return; }
      function run(){
        if(!window.grecaptcha||!window.grecaptcha.ready){ cb(''); return; }
        grecaptcha.ready(function(){
          grecaptcha.execute(siteKey,{action:action||'otp_request'}).then(function(tok){ cb(tok||''); }).catch(function(){ cb(''); });
        });
      }
      if(!window.grecaptcha){
        var sc=document.createElement('script');
        sc.src='https://www.google.com/recaptcha/api.js?render='+encodeURIComponent(siteKey);
        sc.onload=run; sc.onerror=function(){ cb(''); };
        document.head.appendChild(sc);
      } else run();
    }

    function applyDashboardBlockState(ex){
      var box=document.getElementById('app-hard-block');
      var msg=document.getElementById('app-hard-block-msg');
      document.body.classList.remove('app-hard-blocked');
      if(box) box.classList.add('hidden');
      if(!currentTestToken||!ex||ex.error)return;
      if(currentUser&&isBookDeckAdminRole(currentUser.role))return;
      if(!ex.appBlocked)return;
      var lines=[];
      if((ex.loans||[]).some(function(L){return L.overdue;})) lines.push('You have overdue equipment. Please return it to the equipment desk as soon as possible.');
      if((ex.penalties||[]).some(function(p){return p.active && !p.isYellowWarning;})) lines.push('Your account has an active penalty (beyond a yellow warning). Please contact the equipment office.');
      if(!lines.length)return;
      if(msg) msg.textContent=lines.join(' ')+' The app is locked until this is resolved.';
      if(box) box.classList.remove('hidden');
      document.body.classList.add('app-hard-blocked');
    }

      var email=(document.getElementById('otp-email').value||'').trim().toLowerCase();
      var code=(document.getElementById('otp-code').value||'').trim();
      if(!email||!/^\d{6}$/.test(code)){ setLoginMsg(t('msgOtpCodeInvalid'),true); return; }

        apiPublicRequest('/auth/otp/verify', { method: 'POST', body: { email: email, code: code } }).then(function(res){
          btn.disabled=false; btn.innerText=old;
          if(!res||!res.ok){ setLoginMsg((res&&res.error)||t('msgOtpVerifyFailed'),true); return; }
          var profile = res.profile || {};
          var authed = { token: String(res.token || ''), user: mapAuthProfileToCurrentUser_(profile) };
          if (!authed.token) { setLoginMsg(t('msgOtpVerifyFailed'), true); return; }
          currentTestToken = authed.token;
          currentUser = authed.user;
          if (!authed.user.onboarding_completed) {
            showOtpRegisterStep(String(profile.user_type || authed.user.role || 'student') === 'staff' ? 'staff' : 'student');
            return;
          }
          if(applyAuthenticatedSession(authed)) return;
        }).catch(function(err){ btn.disabled=false; btn.innerText=old; setLoginMsg((err&&err.message)||t('msgOtpVerifyFailed'),true); });
      });
    }

    function soundSectionSortRankByKey(sectionKey) {
      var s = String(sectionKey || '').toLowerCase();
      if (!s) return 999;
      // Desired order:
      // wmobmic -> wlavmic -> whandmic -> audirecorder -> shotgun -> headphone -> pole
      if (
        s.indexOf('wmobmic') >= 0 ||
        (s.indexOf('wireless') >= 0 && (s.indexOf('mobile') >= 0 || s.indexOf('mob') >= 0))
      ) return 0;
      if (
        s.indexOf('wlavmic') >= 0 ||
        (s.indexOf('wireless') >= 0 && (s.indexOf('lavalier') >= 0 || s.indexOf('lav') >= 0 || s.indexOf('yaka') >= 0))
      ) return 1;
      if (
        s.indexOf('whandmic') >= 0 ||
        (s.indexOf('wireless') >= 0 && (s.indexOf('hand') >= 0 || s.indexOf('el') >= 0))
      ) return 2;
      if (s.indexOf('audirecorder') >= 0 || s.indexOf('audio recorder') >= 0 || s.indexOf('recorder') >= 0) return 3;
      if (s.indexOf('shotgun') >= 0) return 4;
      if (s.indexOf('heaphone') >= 0 || s.indexOf('headphone') >= 0 || s.indexOf('kulaklık') >= 0 || s.indexOf('kulaklik') >= 0) return 5;
      if (s.indexOf('pole') >= 0 || s.indexOf('boom') >= 0) return 6;
      return 10;
    }
    function findEquipGroupByCartId(cid) {
      var c = String(cid || '').trim();
      if (!c) return null;
      var cu = c.toUpperCase();
      function rowMatch(i) {
        if (String(i.id || '').trim().toUpperCase() === cu) return true;
        if (Array.isArray(i.availableIds) && i.availableIds.some(function(x) { return String(x).trim().toUpperCase() === cu; })) return true;
        if (Array.isArray(i.details) && i.details.some(function(d) { return String(d.equipment_id || '').trim().toUpperCase() === cu; })) return true;
        return false;
      }
      for (var k in equipmentCategoryCache) {
        if (!Object.prototype.hasOwnProperty.call(equipmentCategoryCache, k)) continue;
        var arr = equipmentCategoryCache[k];
        if (!arr || !arr.length) continue;
        for (var j = 0; j < arr.length; j++) {
          if (rowMatch(arr[j])) return arr[j];
        }
      }
      return allEquipmentData.find(rowMatch) || null;
    }

      if (g) return g;
      if (!detailKey || detailKey.indexOf('eq-detail-') !== 0) return null;
      var slug = detailKey.replace(/^eq-detail-/, '');
      for (var k in equipmentCategoryCache) {
        if (!Object.prototype.hasOwnProperty.call(equipmentCategoryCache, k)) continue;
        var arr = equipmentCategoryCache[k];
        if (!arr) continue;
        for (var i = 0; i < arr.length; i++) {
          if (detailKeyForEquipmentGroup(arr[i]) === detailKey) return arr[i];
          var gk = arr[i].groupKey ? String(arr[i].groupKey).toLowerCase().replace(/[^a-z0-9]+/g, '-') : '';
          if (gk && gk === slug) return arr[i];
        }
      }
      return null;
    }

    function isPublicHoliday(date){
      var dt = (date instanceof Date) ? date : new Date(String(date || ''));
      if (isNaN(dt.getTime())) return false;
      var d=dt.getDate(),m=dt.getMonth()+1,y=dt.getFullYear();
      if(d===1&&m===1)return true;if(d===23&&m===4)return true;if(d===1&&m===5)return true;if(d===19&&m===5)return true;
      if(d===15&&m===7)return true;if(d===30&&m===8)return true;if(d===28&&m===10)return true;if(d===29&&m===10)return true;
      var ds=y+'-'+String(m).padStart(2,'0')+'-'+String(d).padStart(2,'0');

    function renderStudioBusyInto(container, studioId, res) {
      if (!container) return;
      if (!res || res.error) {
        container.innerHTML = '<p class="text-xs text-red-600 p-2">' + escHtml((res && res.error) ? res.error : 'Error') + '</p>';
        return;
      }
      if (res.noCalendar) {
        container.innerHTML = '<p class="text-xs text-gray-500 text-center p-4">' + (currentLang === 'tr' ? 'Takvim yapılandırılmamış.' : 'Calendar not configured.') + '</p>';
        return;
      }
      var blocks = res.blocks || [];
      var rangeStart = Number(res.rangeStartMs) || 0;
      var stud = (studiosLoadCache || []).find(function(x) { return x.id === studioId; });
      var weekendOk = stud ? !!stud.weekendOk : false;
      var pad2 = function(n) { return String(n).padStart(2, '0'); };
      var base = new Date(rangeStart);
      if (isNaN(base.getTime())) base = new Date();
      base.setHours(0, 0, 0, 0);
      var days = [];
      var i;
      for (i = 0; i < 8; i++) {
        days.push(new Date(base.getFullYear(), base.getMonth(), base.getDate() + i));
      }
      var slots = [];
      for (var hm = 9 * 60; hm <= 16 * 60 + 30; hm += 30) {
        if (hm >= 12 * 60 && hm < 13 * 60) continue;
        slots.push(hm);
      }
      function dayKind(dayDate) {
        if (safeIsPublicHoliday(dayDate)) return 'closed';
        var wd = dayDate.getDay();
        if (!weekendOk && (wd === 0 || wd === 6)) return 'closed';
        return 'open';
      }
      function slotState(dayDate, startMin) {
        if (dayKind(dayDate) !== 'open') return 'closed';
        var y = dayDate.getFullYear(), mo = dayDate.getMonth(), dd = dayDate.getDate();
        var sh = Math.floor(startMin / 60), smi = startMin % 60;
        var sm = new Date(y, mo, dd, sh, smi, 0, 0).getTime();
        var em = sm + 30 * 60000;
        for (i = 0; i < blocks.length; i++) {
          var b = blocks[i];
          if (b.startMs < em && b.endMs > sm) return 'busy';
        }
        return 'free';
      }
      var wk = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
      var th = '<tr><th class="p-1 w-10"></th>' + days.map(function(day) {
        return '<th class="p-1 text-[8px] md:text-[10px] font-bold text-gray-600 border-b border-gray-200 min-w-[36px]">' + pad2(day.getDate()) + '.' + pad2(day.getMonth() + 1) + '<br><span class="text-gray-400 font-semibold">' + wk[day.getDay()] + '</span></th>';
      }).join('') + '</tr>';
      var rows = slots.map(function(startMin) {
        var sh = Math.floor(startMin / 60), smi = startMin % 60;
        var timeL = pad2(sh) + ':' + pad2(smi);
        var cells = days.map(function(day) {
          var st = slotState(day, startMin);
          var bg = st === 'free' ? 'bg-emerald-100' : (st === 'busy' ? 'bg-rose-200' : 'bg-gray-200');
          var title = st === 'free' ? 'Available' : (st === 'busy' ? 'Busy' : 'Closed');
          return '<td class="p-0.5 border-b border-gray-100"><div class="' + bg + ' h-4 md:h-5 rounded-sm" title="' + title + '"></div></td>';
        }).join('');
        return '<tr><td class="pr-1 py-0.5 text-[8px] md:text-[10px] text-gray-500 font-mono whitespace-nowrap">' + timeL + '</td>' + cells + '</tr>';
      }).join('');
      var legFree = currentLang === 'tr' ? 'Uygun' : 'Available';
      var legBusy = currentLang === 'tr' ? 'Meşgul' : 'Busy';
      var legClosed = currentLang === 'tr' ? 'Kapalı' : 'Closed';
      container.innerHTML =
        '<div class="overflow-x-auto max-h-[min(50vh,360px)]">' +
        '<table class="w-full border-collapse text-center">' +
        '<thead>' + th + '</thead>' +
        '<tbody>' + rows + '</tbody>' +
        '</table></div>' +
        '<p class="text-[9px] text-gray-500 mt-2 flex flex-wrap gap-3">' +
        '<span><span class="inline-block w-3 h-3 rounded-sm bg-emerald-100 align-middle mr-1"></span> ' + legFree + '</span>' +
        '<span><span class="inline-block w-3 h-3 rounded-sm bg-rose-200 align-middle mr-1"></span> ' + legBusy + '</span>' +
        '<span><span class="inline-block w-3 h-3 rounded-sm bg-gray-200 align-middle mr-1"></span> ' + legClosed + '</span>' +
        '</p>';
    }

    function formatMyDate(dateStr){
      if(!dateStr)return '';
      try{
        var raw=String(dateStr).trim();
        // Keep sheet/server wall-clock time without timezone shift in UI.
        var m=raw.match(/^(\d{4})-(\d{2})-(\d{2})[T\s](\d{2}):(\d{2})/);
        if(m){
          return m[3]+'.'+m[2]+'.'+m[1]+' / '+m[4]+'.'+m[5];
        }
        var d=new Date(raw);
        if(isNaN(d.getTime()))return raw;
        var pad=function(n){return String(n).padStart(2,'0');};
        return pad(d.getDate())+'.'+pad(d.getMonth()+1)+'.'+d.getFullYear()+' / '+pad(d.getHours())+'.'+pad(d.getMinutes());
      }catch(e){return dateStr;}
    }

          return {
            id: id,
            displayId: code || id,

            notifyGroupKey: id,
            name: name,
            typeLabel: typeLabel,
            category: typeLabel,
            status: statusRaw || 'UNKNOWN',
            image: photo,
            returnDate: ret,
            eq_level: levelRaw,
            eq_max_per_reservation: perResCap,
            details: [{
              equipment_id: id,

    function groupEquipmentItemsByName(items){
      var src=Array.isArray(items)?items:[];
      var map={};
      function isAvailStatus(s){
        var u=String(s||'').toUpperCase();
        return u.indexOf('AVAIL')>=0||u==='MÜSAİT'||u==='UYGUN';
      }
      src.forEach(function(item){
        var details=Array.isArray(item.details)?item.details:[];
        if(!details.length){
          var n0=String(item.name||item.typeLabel||item.id||'').trim();
          if(!n0) return;
          details=[{equipment_id:String(item.id||'').trim(),equipment_code:String(item.displayId||item.id||'').trim(),name:n0,status:String(item.status||''),returnDate:String(item.returnDate||'').trim(),returnDateMs:null,conditionIn:'–',photo:String(item.image||'').trim()}];
        }
        details.forEach(function(d){
          var unitName=String(d.name||item.name||'').trim();
          if(!unitName) return;
          var parentTypeLabel = String(item.typeLabel || item.category || '').trim();
          var key=(String(getBaseCategory(item)||'All').toLowerCase()+'::'+unitName.toLowerCase());
          if(!map[key]){
            map[key]={
              id:String(item.id||d.equipment_id||'').trim(),
              groupKey:'nm-'+key.replace(/[^a-z0-9]+/g,'-'),
              notifyGroupKey:String(item.notifyGroupKey||item.groupKey||'').trim(),
              name:unitName,
              typeLabel:parentTypeLabel || unitName,
              modelNames:[unitName],
              category:item.category,
              status:'Checked Out',
              image:String(d.photo||item.image||'').trim(),
              stock:0,outCount:0,totalCount:0,returnDate:'',returnSchedule:[],availableIds:[],
              eq_level:item.eq_level,eq_max_per_reservation:item.eq_max_per_reservation,details:[],
              earliestReturnMs:null,earliestReturnLabel:'',earliestReturnDateOnly:''
            };
          }
          var g=map[key];
          if(!g.notifyGroupKey&&item.groupKey) g.notifyGroupKey=String(item.groupKey);
          if(!g.image){
            var p=String(d.photo||item.image||'').trim();
            if(p) g.image=p;
          }
          g.totalCount+=1;
          var id=String(d.equipment_id||'').trim();
          var stRaw = String(d.status || '').trim();
          var hasDetailStatus = !!stRaw;
          var avFromDetail = isAvailStatus(stRaw);
          var avFromList = (id&&Array.isArray(item.availableIds)&&item.availableIds.some(function(a){return String(a).toUpperCase()===id.toUpperCase();}));
          // Per-unit status is authoritative when present.
          var av = hasDetailStatus ? avFromDetail : avFromList;
          if(av){
            g.stock+=1;
            g.status='Available';
            if(id) g.availableIds.push(id);
          }else{
            g.outCount+=1;
            var rms = (d.returnDateMs != null && !isNaN(Number(d.returnDateMs))) ? Number(d.returnDateMs) : NaN;
            if (!isNaN(rms) && (g.earliestReturnMs == null || rms < g.earliestReturnMs)) {
              g.earliestReturnMs = rms;
              g.earliestReturnLabel = String(d.returnDate || '').trim();
            } else if (!g.earliestReturnLabel && d.returnDate) {
              g.earliestReturnLabel = String(d.returnDate || '').trim();
            }
          }
          g.details.push({
            equipment_id:id,
            equipment_code:String(d.equipment_code || item.displayId || id || '').trim(),
            name:unitName,
            status:d.status,
            returnDate:d.returnDate,
            returnDateMs:d.returnDateMs,
            conditionIn:d.conditionIn,
            photo:d.photo
          });
        });
      });
      return Object.keys(map).map(function(k){return map[k];});
    }
    function bookingSpanDays(isoStart, isoEnd) {
      var a = new Date(isoStart), b = new Date(isoEnd);
      if (isNaN(a.getTime()) || isNaN(b.getTime())) return 0;
      return (b - a) / 86400000;
    }
    function equipmentHolidayDatesInRange_(startIso, endIso) {
      var s = new Date(String(startIso || ''));
      var e = new Date(String(endIso || ''));
      if (isNaN(s.getTime()) || isNaN(e.getTime()) || e <= s) return [];
      var cur = new Date(s.getTime());
      cur.setHours(0, 0, 0, 0);
      var last = new Date(e.getTime() - 1);
      last.setHours(0, 0, 0, 0);
      var out = [];
      while (cur <= last) {
        if (safeIsPublicHoliday(cur)) {
          out.push(String(cur.getDate()).padStart(2, '0') + '.' + String(cur.getMonth() + 1).padStart(2, '0') + '.' + cur.getFullYear());
        }
        cur.setDate(cur.getDate() + 1);
      }
      return out;
    }

    function buildProvisionalStudioEndIso(sD, sT) {
      var t = (sT && String(sT).length === 5) ? sT + ':00' : (sT || '10:00:00');
      var s = new Date(sD + 'T' + t);
      if (isNaN(s.getTime())) return '';
      var e = new Date(s.getTime() + 2 * 60 * 60 * 1000);
      var cap = new Date(sD + 'T21:00:00');
      if (!isNaN(cap.getTime()) && e > cap) e = cap;
      var pad = function(n) { return String(n).padStart(2, '0'); };
      return e.getFullYear() + '-' + pad(e.getMonth() + 1) + '-' + pad(e.getDate()) + 'T' + pad(e.getHours()) + ':' + pad(e.getMinutes());
    }
    function runStudioCalendarCheck() {
      if (!activeStudioId || typeof google === 'undefined' || !google.script || !google.script.run) {
        hideStudioCalendarWarning();
        return;
      }
      var sD = document.getElementById('studio-start-date').value;
      var sT = document.getElementById('studio-start-time').value;

        hideStudioCalendarWarning();

      var startIso = sD + 'T' + sT;
      var endIso = (eD && eT) ? (eD + 'T' + eT) : buildProvisionalStudioEndIso(sD, sT);
      if (!endIso) { hideStudioCalendarWarning(); return; }
      var sMs = new Date(startIso).getTime();
      var eMs = new Date(endIso).getTime();
      if (isNaN(sMs) || isNaN(eMs) || eMs <= sMs) {
        hideStudioCalendarWarning();
        return;
      }
      var seq = ++studioCalCheckSeq;
      studioCalendarBusy = false;
      studioCalendarChecking = true;
      var w = document.getElementById('studio-calendar-warning');
      if (w) w.classList.add('hidden');
      updateStudioPdfLock();
      google.script.run
        .withSuccessHandler(function(res) {
          if (seq !== studioCalCheckSeq) return;
          studioCalendarChecking = false;
          if (!res) { lastStudioCalBusyResult = false; hideStudioCalendarWarning(); return; }
          if (res.error) {
            lastStudioCalBusyResult = false;
            showStudioCalendarErrorBanner(res.error);
            return;
          }
          if (res.skipped) { lastStudioCalBusyResult = false; hideStudioCalendarWarning(); return; }
          if (res.busy) {
            var msgBusy = currentLang === 'tr' ? STUDIO_CALENDAR_BUSY_MSG_TR : STUDIO_CALENDAR_BUSY_MSG_EN;

              showAlert(currentLang === 'tr' ? 'Stüdyo müsait değil' : 'Studio unavailable', msgBusy, true);
            }
            return;
          }
          lastStudioCalBusyResult = false;
          hideStudioCalendarWarning();
        })
        .withFailureHandler(function(err) {
          if (seq !== studioCalCheckSeq) return;
          studioCalendarChecking = false;
          lastStudioCalBusyResult = false;
          showStudioCalendarErrorBanner(err && err.message ? err.message : 'Calendar check failed.');
        })

    function formatRemainMs(ms) {
      if (ms == null || isNaN(ms)) return '';
      if (ms < 0) {
        var a = Math.abs(ms);
        var h = Math.floor(a / 3600000);
        if (h >= 48) return '(' + Math.floor(h / 24) + ' days overdue)';
        if (h >= 1) return '(' + h + ' h overdue)';
        return '(' + Math.max(1, Math.floor(a / 60000)) + ' min overdue)';
      }
      var h = Math.floor(ms / 3600000);
      var d = Math.floor(h / 24);
      if (d >= 1) return '(' + d + 'd ' + (h % 24) + 'h left)';
      if (h >= 1) return '(' + h + 'h left)';
      return '(' + Math.max(1, Math.floor(ms / 60000)) + ' min left)';
    }

      var raw = String(url || '').trim();

      var base = raw.split('?')[0];
      var query = hasQuery ? raw.substring(raw.indexOf('?') + 1) : '';
      var parts = query ? query.split('&').filter(function(p){

      }) : [];
      parts.push('mode=AGENDA');
      parts.push('wkst=2');
      return base + '?' + parts.join('&');

    function mondayIsoFrom_(d){
      var x = d instanceof Date ? new Date(d.getTime()) : new Date(String(d || ''));
      if (isNaN(x.getTime())) x = new Date();
      x.setHours(0,0,0,0);
      var day=x.getDay();
      var diff=(day===0?-6:(1-day));
      x.setDate(x.getDate()+diff);
      var y=x.getFullYear(),m=String(x.getMonth()+1).padStart(2,'0'),dd=String(x.getDate()).padStart(2,'0');
      return y+'-'+m+'-'+dd;
    }

    function fetchStudioWeekGridViaApi_(studioId, weekStartIso){
      var startIso = String(weekStartIso || '').trim();
      if(!startIso) startIso = mondayIsoFrom_(new Date());
      var startAt = startIso + 'T00:00:00';
      var endDate = new Date(startIso + 'T00:00:00');
      endDate.setDate(endDate.getDate() + 6);
      var ey = endDate.getFullYear(), em = String(endDate.getMonth()+1).padStart(2,'0'), ed = String(endDate.getDate()).padStart(2,'0');
      var endIso = ey + '-' + em + '-' + ed + 'T23:59:59';
      return apiRequest('/studio-reservations?from=' + encodeURIComponent(startAt) + '&to=' + encodeURIComponent(endIso), { method: 'GET' })
        .then(function(payload){
          var rows = (payload && payload.data) ? payload.data : [];
          var days = [];
          for (var i=0;i<7;i++){
            var d = new Date(startIso + 'T00:00:00');
            d.setDate(d.getDate() + i);
            days.push({
              iso: d.getFullYear() + '-' + String(d.getMonth()+1).padStart(2,'0') + '-' + String(d.getDate()).padStart(2,'0'),
              day: d.getDate(),
              month: d.getMonth()+1,
              dow: d.getDay()
            });
          }
          var slots = [];
          for (var hm = 9 * 60; hm <= 20 * 60; hm += 60) {
            var hh = String(Math.floor(hm / 60)).padStart(2, '0');
            var mm = String(hm % 60).padStart(2, '0');
            slots.push({ mins: hm, label: hh + ':' + mm });
          }
          var cells = {};
          var cellBusyStart = {};
          var studioMeta = getStudioMetaById_(studioId);
          var studioKey = resolveStudioCanonicalKey_(studioMeta || {}) || String(studioId || '').trim().toUpperCase();
          var weekendOk = !!(studioMeta && studioMeta.weekendOk);
          if (studioKey === 'GREEN') weekendOk = true;
          days.forEach(function(d){
            var isClosedDay = safeIsPublicHoliday(new Date(d.iso + 'T00:00:00')) || (!weekendOk && d.dow === 0);
            slots.forEach(function(s){
              var key = d.iso + '|' + s.mins;
              cells[key] = isClosedDay ? 'closed' : 'free';
            });
          });
          var activeStatuses = { pending:1, approved:1, checked_out:1, picked_up:1, key_out:1 };
          var sid = String(studioId || '').trim().toUpperCase();
          rows.forEach(function(r){
            var rid = String((r && r.studio_id) || '').trim().toUpperCase();
            if(!rid || rid !== sid) return;
            var st = String((r && r.status) || '').trim().toLowerCase();
            if(!activeStatuses[st]) return;
            var sMs = new Date(String(r.start_at || '')).getTime();
            var eMs = new Date(String(r.end_at || '')).getTime();
            if(isNaN(sMs) || isNaN(eMs) || eMs <= sMs) return;
            days.forEach(function(d){
              slots.forEach(function(s){
                var slotMs = new Date(d.iso + 'T' + s.label + ':00').getTime();
                var slotEnd = slotMs + 60 * 60000;
                if(slotMs < eMs && slotEnd > sMs){
                  var key = d.iso + '|' + s.mins;
                  if(cells[key] === 'free'){
                    cells[key] = 'busy';
                    cellBusyStart[key] = true;
                  }
                }
              });
            });
          });
          return { days: days, slots: slots, cells: cells, cell_busy_start: cellBusyStart, week_start: startIso };
        });
    }

    function formatPenaltyRemainMs(ms) {
      if (ms == null || isNaN(ms)) return '';
      var d = Math.floor(ms / 86400000);
      var h = Math.floor((ms % 86400000) / 3600000);
      if (d >= 1) return d + 'd ' + h + 'h remaining';
      if (h >= 1) return h + 'h remaining';
      return Math.max(0, Math.floor(ms / 60000)) + ' min remaining';
    }
    function normalizeBookingStatus_(s) {
      return String(s || '').toLowerCase().replace(/[_-]+/g, ' ').replace(/\s+/g, ' ').trim();
    }

    function loadMyBookings(){
      var tbody=document.getElementById('bookings-active-body');
      var loansEl=document.getElementById('bookings-loans');
      var penEl=document.getElementById('bookings-penalties');
      var extrasLoans=[];
      var bookingLoanFallback=[];
      function renderLoansPanel_() {
        if(!loansEl) return;
        var loans = extrasLoans.length ? extrasLoans : bookingLoanFallback;
        if(!loans.length){
          loansEl.innerHTML='<p class="text-gray-400 text-sm">No checked-out equipment on your account.</p>';
          return;
        }
        loansEl.innerHTML='<ul class="space-y-3">' + loans.map(function(L){
          var warn=L.overdue?'<span class="text-red-600 font-black ml-1" title="Overdue">⚠</span>':'';
          var timePart=formatRemainMs(L.msLeft);
          var ext='';
          if(bookingLimits.canExtendActiveLoan&&L.resGroupId&&L.eqId&&L.handover){
            ext='<button type="button" class="extend-row-btn text-[10px] font-bold text-[#e30613] hover:underline ml-2" data-group="'+escAttr(L.resGroupId)+'" data-eqid="'+escAttr(L.eqId)+'" data-start="'+escAttr(L.handover)+'" data-end="'+escAttr(L.resEnd||L.due)+'">Extend</button>';
          }
          return '<li class="flex flex-wrap items-baseline justify-between gap-2 border-b border-gray-100 pb-2 last:border-0"><span class="font-bold text-gray-900 text-sm">'+escHtml(L.name)+' <span class="text-[10px] font-mono text-gray-400">'+escHtml(L.eqDisplayId||L.eqId)+'</span>'+warn+ext+'</span><span class="text-xs '+(L.overdue?'text-red-600 font-bold':'text-gray-500')+'">Due '+escHtml(formatMyDate(L.due))+(timePart?' <span class="text-gray-500">'+timePart+'</span>':'')+'</span></li>';
        }).join('')+'</ul>';
      }
      if(tbody)tbody.innerHTML='<tr><td colspan="5" class="p-6 text-center text-blue-500 text-xs animate-pulse">Loading…</td></tr>';
      if(loansEl)loansEl.innerHTML='<p class="text-gray-400 text-xs animate-pulse">Loading…</p>';
      if(penEl)penEl.innerHTML='<p class="text-gray-400 text-xs animate-pulse">Loading…</p>';
      if (typeof google !== 'undefined' && google.script && google.script.run) {
        google.script.run.withSuccessHandler(function(ex){
          applyDashboardBlockState(ex);
          extrasLoans=(ex&&ex.loans)||[];
          var penalties=(ex&&ex.penalties)||[];
          if(ex&&ex.error&&(!extrasLoans||!extrasLoans.length)&&(!bookingLoanFallback||!bookingLoanFallback.length)){
            if(loansEl)loansEl.innerHTML='<p class="text-red-500 text-xs">'+(ex.error)+'</p>';
          } else {
            renderLoansPanel_();
          }
          if(penEl){
            var activePen=(penalties||[]).filter(function(p){return p.active;});
            if(!activePen.length){
              penEl.innerHTML='<p class="text-emerald-800 font-semibold text-sm md:text-base leading-relaxed">👏 Thank you for returning equipment on time — you have no penalties on your account. Keep it up!</p>';

              }).join('');

          }
        }).withFailureHandler(function(){

          else if(loansEl)loansEl.innerHTML='<p class="text-red-500 text-xs">Could not load loans.</p>';

      }

      apiRequest('/me/bookings', { method: 'GET' }).then(function(payload){
        var data = (payload && payload.data) ? payload.data : [];
        if(!tbody)return;
        var nowMs = Date.now();
        bookingLoanFallback=(data||[]).filter(function(b){
          var st=normalizeBookingStatus_(b.status);

        }).map(function(b){
